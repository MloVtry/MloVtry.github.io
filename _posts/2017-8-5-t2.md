#include<iostream>
#include<cstdio>
#define mod 998244353
#define M 999000000-2
#define ll long long
using namespace std;
ll T,n;
ll c[100021],it[100021];
ll last[100021];
void get_c()
{
    c[2]=1;
    for(ll i=3;i<=100020;i++)
    {
        c[i]=(i-1)*i/2;
        c[i]%=mod;
    }
    it[3]=2;
    for(ll i=4;i<=100020;++i)
    {
        it[i]=it[i-1]*(i-2)+1;
        it[i]%=mod;
    }
    last[3]=1;
    for(ll i=4;i<=100020;++i)
    {
        last[i]=c[i-1]*it[i-1]+last[i-1];
        last[i]%=mod;
    }
}
ll cb[1000010],lastb[1000010],itb[1000010];
void get_b()
{
	cb[0]=1420232,lastb[0]=876466444,itb[0]=141309211;
	ll to=1000005;
	for(ll i=1;i<=1000005;++i)
	{
		ll num=i+M;
		cb[i]=cb[i-1]+num-1;
		cb[i]%=mod;
		lastb[i]=cb[i-1]*itb[i-1]+lastb[i-1];
		lastb[i]%=mod;
		itb[i]=itb[i-1]*(num-2)+1;
		itb[i]%=mod;
	}
}
int main()
{
	scanf("%lld",&T);
	get_c();
	get_b();
	while(T--)
	{
		scanf("%lld",&n);
		if(n>M)
		{
			n%=M;
			ll ans=cb[n-2]*itb[n-1]+lastb[n-1];
			ans%=mod;
			printf("%lld\n",ans);
			continue;
		}
		ll ans=c[n-2]*it[n-1]+last[n-1];
		ans%=mod;
		printf("%lld\n",ans);
	}
}
