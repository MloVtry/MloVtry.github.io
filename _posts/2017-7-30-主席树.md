#include<iostream>
#include<cstdio>
#include<algorithm>
#define TN 9000000
#define N 300001
#define M 500001
using namespace std;
int n,m;
int s[TN],ls[TN],rs[TN];
int knt,root[N],a[N],b[N];
using namespace std;
void buid(int A,int &B,int l,int r,int it)
{
	int mid=(l+r)>>1;
	B=++knt;
	s[B]=s[A]+1;
	if(l==r) return;
	if(it<=mid)	buid(ls[A],ls[B],l,mid,it);
	else buid(rs[A],rs[B],mid+1,r,it);
	if(!ls[B]) ls[B]=ls[A];
	if(!rs[B]) rs[B]=rs[A];
}
int ask(int V,int U,int l,int r,int k)
{
	if(l==r) return r;
	int mid=(l+r)>>1;
	if(s[ls[V]]-s[ls[U]]>=k) return ask(ls[V],ls[U],l,mid,k);
	else return ask(rs[V],rs[U],mid+1,r,k-s[ls[V]]+s[ls[U]]);
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;++i)
	{
		scanf("%d",&a[i]);
		b[i]=a[i];
	}
	sort(b+1,b+n+1);
	for(int i=1;i<=n;++i)
	{
		int it=lower_bound(b+1,b+n+1,a[i])-b;
		buid(root[i-1],root[i],1,n,it);
	}
	for(int i=1;i<=m;++i)
	{
		int u,v,k;
		scanf("%d%d%d",&u,&v,&k);
		printf("%d\n",b[ask(root[v],root[u-1],1,n,k)]);
	}
	return 0;
}
