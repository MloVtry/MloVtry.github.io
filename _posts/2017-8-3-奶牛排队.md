#include<iostream>
#include<cstdio>
using namespace std;
int n,say,top,ans;
struct node
{
	int pos,max,l;
}s[100010];
int read() 
{
    int out=0,flag=1;
    char c=getchar();
    while(c<48||c>57) {if(c=='-') flag=-1;c=getchar();}
    while(c>=48&&c<=57)
    {
        out=out*10+c-48;
        c=getchar();
    }
    return out*flag;
}
int main()
{
	freopen("tahort.in","r",stdin);
	freopen("tahort.out","w",stdout);
	n=read();
	say=read();
	s[++top]=(node){1,say,say};
	for(int i=2;i<=n;++i)
	{
		say=read();
		while(top&&s[top].l>say) top--;
		s[++top]=(node){i,say,say};
		while(top&&s[top].max<=say) top--; 
		s[++top].max=say;
		ans=max(ans,i-s[top].pos+1);
	}
	if(ans==1) ans=0;
	printf("%d",ans);
	return 0;
}
